// --- Dados Principais ---
  - userId: string (NN)
  - setupTitle: string (NN)
  - gameVersion: string (NN)
  - car: string (NN)
  - track: string (NN)
  - condition: string (NN)
  - setupType: string (NN)
  - controlType: string (NN)
  - notes: string (Nullable)
  - setupDetails: map { ... } // (Contém as configurações do setup)

  // --- Campos de Organização e Referência ---
  - folderId: string (Nullable)       // ID do documento da coleção `folders`
  - originalSetupId: string (Nullable) // Se for um clone, guarda o ID do setup original

  // --- Campos de Status e Comunidade ---
  - isPublic: boolean (NN)
  - createdAt: timestamp (NN)
  - updatedAt: timestamp (NN)
  - ratingCount: number (NN, default: 0) // Quantidade de avaliações recebidas
  - avgRating: number (NN, default: 0)   // Média das avaliações (de 1 a 5, por exemplo)
  - cloneCount: number (NN, default: 0)  // Quantas vezes foi salvo por outros usuários
  
  // --- Sub-coleções ---
  /setups/{setupId}/comments/{commentId}
    - userId: string (NN)
    - userName: string (NN) // Denormalizado para não precisar buscar na coleção 'users'
    - userProfilePic: string (Nullable) // Denormalizado também
    - text: string (NN)
    - createdAt: timestamp (NN)
    
  /setups/{setupId}/performanceLogs/{logId}
    - lapTime: string (NN) // Ex: "1:28.452"
    - condition: string (Nullable) // Ex: "Seco/Chuva/Chuva forte"
    - notes: string (Nullable)
    - loggedAt: timestamp (NN)
